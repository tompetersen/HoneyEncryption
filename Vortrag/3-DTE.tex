\begin{frame}[t]
	\frametitle{DTE}

	Bekannt sein muss:
	\begin{itemize}
		\item Menge/Struktur der Nachrichten
		\begin{itemize}
			\item endlich speicherbar/berechenbar
			\item unendlich berechenbar
		\end{itemize}
		\item Verteilung der Nachrichten
		\begin{itemize}
			\item Nachricht wahrscheinlicher $\Rightarrow$ mehr Seeds
		\end{itemize}
	\end{itemize}
